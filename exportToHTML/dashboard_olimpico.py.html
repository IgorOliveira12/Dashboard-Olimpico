# Carregamento dos dados
@st.cache_data
def carregar_dados(caminho):
    df = pd.read_csv(caminho)
    df = df.dropna(subset=["Age", "Height", "Weight", "Sex", "Medal"])
    df["BMI"] = df["Weight"] / (df["Height"] / 100) ** 2
    return df

df = carregar_dados("data/athlete_events.csv")

# Aprendizagem Autom√°tica
with tab5:
    st.markdown("ü§ñ **Modelos para Prever Tipo de Medalha**")

    df_ml = df_filtro[df_filtro["Medal"].isin(["Gold", "Silver", "Bronze"])]
    features = ["Age", "Height", "Weight", "BMI"]
    X = df_ml[features]
    y = LabelEncoder().fit_transform(df_ml["Medal"])
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

    modelos = {
        "Regress√£o Log√≠stica": LogisticRegression(max_iter=1000),
        "Naive Bayes": GaussianNB(),
        "√Årvore de Decis√£o": DecisionTreeClassifier(),
        "Random Forest": RandomForestClassifier(n_estimators=100),
        "K-Nearest Neighbors (KNN)": KNeighborsClassifier(n_neighbors=5)
    }

    for nome, modelo in modelos.items():
        modelo.fit(X_train, y_train)
        acc = modelo.score(X_test, y_test)
        st.markdown(f"### üìå {nome}: accuracy = {acc:.2f}")
        y_pred = modelo.predict(X_test)

        st.markdown("**Relat√≥rio de Classifica√ß√£o:**")
        st.text(classification_report(y_test, y_pred))

        cm = confusion_matrix(y_test, y_pred)
        fig_cm, ax = plt.subplots(figsize=(4, 3))
        sns.heatmap(cm, annot=True, fmt='d', cmap='Oranges', ax=ax)
        ax.set_title(f"Matriz de Confus√£o - {nome}")
        ax.set_xlabel("Previsto")
        ax.set_ylabel("Real")
        st.pyplot(fig_cm)

    # PCA
    st.markdown("### üîç PCA - An√°lise de Componentes Principais")
    pca = PCA(n_components=2)
    X_pca = pca.fit_transform(X)
    df_pca = pd.DataFrame(X_pca, columns=["PC1", "PC2"])
    df_pca["Medalha"] = df_ml["Medal"].values
    fig_pca = px.scatter(df_pca, x="PC1", y="PC2", color="Medalha", title="Proje√ß√£o PCA das Medalhas")
    st.plotly_chart(fig_pca, use_container_width=False, width=600)

    # Valida√ß√£o cruzada
    st.markdown("### üß™ Valida√ß√£o Cruzada com KNN")
    knn = KNeighborsClassifier()
    scores = cross_val_score(knn, X, y, cv=5)
    st.write(f"Acuracys por fold: {scores}")
    st.write(f"Acuracy m√©dia: {scores.mean():.2f}")

    # Previs√£o interativa
    st.markdown("### üßæ Previs√£o Interativa")
    with st.form("form_predicao"):
        st.write("Introduza os dados do atleta para prever a medalha:")
        idade = st.number_input("Idade", 10, 100, 25)
        altura = st.number_input("Altura (cm)", 100, 250, 175)
        peso = st.number_input("Peso (kg)", 30, 200, 70)
        bmi = peso / (altura / 100) ** 2
        modelo_sel = st.selectbox("Modelo", list(modelos.keys()))
        submitted = st.form_submit_button("Prever")

        if submitted:
            modelo = modelos[modelo_sel]
            entrada = pd.DataFrame([[idade, altura, peso, bmi]], columns=features)
            pred = modelo.predict(entrada)[0]
            medalha = {0: "Bronze", 1: "Gold", 2: "Silver"}.get(pred, "Desconhecida")
            st.success(f"üèÖ Medalha prevista: **{medalha}**")
    with st.expander("‚ÑπÔ∏è Ajuda: Como interpretar os gr√°ficos e m√©tricas"):
        st.markdown(""" 
        ### üß† Guia R√°pido: Aprendizagem Autom√°tica 
 
        **üî¢ Acur√°cia (accuracy):**   
        Percentagem de previs√µes corretas. Exemplo: `0.85` significa 85% de acerto. 
 
        **üìã Relat√≥rio de Classifica√ß√£o:**   
        - **Precis√£o (Precision):** De todas as previs√µes feitas para uma classe, quantas estavam corretas?   
        - **Revoca√ß√£o (Recall):** De todos os exemplos reais dessa classe, quantos foram identificados corretamente?   
        - **F1-score:** M√©dia ponderada entre precis√£o e revoca√ß√£o. 
 
        **üî≥ Matriz de Confus√£o:**   
        - Mostra os acertos e erros por tipo de medalha. 
        - Diagonal = acertos.   
        - Fora da diagonal = erros (ex: previu "Gold" mas era "Bronze"). 
 
        **üìâ PCA (An√°lise de Componentes Principais):**   
        - Reduz os dados para 2 dimens√µes para visualiza√ß√£o. 
        - Pontos pr√≥ximos e da mesma cor = modelo pode distinguir bem. 
        - Tudo misturado = os dados s√£o dif√≠ceis de separar. 
 
        **üîÅ Valida√ß√£o Cruzada:**   
        - Mede a consist√™ncia do modelo em diferentes divis√µes dos dados. 
        - Mostra as acur√°cias por tentativa (fold) e a m√©dia final. 
 
        **ü§ñ Previs√£o Interativa:**   
        - Introduz idade, altura e peso de um atleta. 
        - O modelo estima a medalha prov√°vel com base nesses dados. 
        """)
